class SoulExplosionSmall : Actor
{
	int soulexplosiondamage;
	int soulexplosionradius;
	
	property ExplosionDamage:soulexplosiondamage;
	property ExplosionRadius:soulexplosionradius;

	Default
	{
		RenderStyle "Add";
		Alpha 1.0;
		Scale 0.75;
		DamageType "Soul";
		SoulExplosionSmall.ExplosionDamage 64;
		SoulExplosionSmall.ExplosionRadius 128;
		-RANDOMIZE;
		+FORCERADIUSDMG;
		+FORCEXYBILLBOARD;
		+NOBLOCKMAP;
		+NOGRAVITY;
	}
	States
	{
	Spawn:
		SOLX A 3 Bright;
		SOLX B 3 Bright A_PlaySound("weapons/casterboom");
		SOLX C 3 Bright A_Explode(soulexplosiondamage, soulexplosiondamage, 0);
		SOLX DEFGH 3 Bright;
		Stop;
	}
}
